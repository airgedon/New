#include <stdlib.h>
void main() {
    float *nums;
    int n,i;
    printf("Read how many numbers: \n");
    scanf("%d", &n);
    nums=(float*)calloc(n,sizeof(float));
    for(i=0;i<n;i++) {
        printf("Please enter number for position %d: ", i+1);
        scanf("%f", &nums[i]);
    }
    printf("The numbers are: \n");
    for(i=0; i<n;i++)
        printf("%f\t", nums[i]);
}
